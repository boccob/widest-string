var currentDate, dayNumber, initDate, monthNames, pad, weekDayNames, i, results;
weekDayNames = ['dom.', 'lun.', 'mar.', 'mié.', 'jue.', 'vie.', 'sáb.'];
monthNames = ['janv.', 'févr.', 'mars', 'avril', 'mai', 'juin', 'juil.', 'août', 'sept.', 'oct.', 'nov.', 'déc.'];
pad = function(n) {
    return ('0' + n).slice(-2);
};
initDate = new Date(2012, 0, 1);
results = [];
for (dayNumber = i = 0; i <= 355; dayNumber = ++i) {
    currentDate = new Date(initDate.getTime() + 1000 * 60 * 60 * 24 * dayNumber);
    results.push("" + weekDayNames[currentDate.getDay()] + " " + (pad(currentDate.getDate())) + " " + monthNames[currentDate.getMonth()]);
}
return results;
